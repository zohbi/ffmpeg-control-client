ffmpeg-control-client by zohbi

The is the client program that send the parameters to server

#!/usr/bin/env python import dbus service = "ch.cett.misse.ffmpeg" ffmpeg = dbus.Interface(dbus.SessionBus().get_object(service,"/"), service) print 'Available commands are: \n' print '1- start \n' print '2- stop \n' print '3- image_size_set \n' print '4- image_size_get \n' print '5- image_rate_set \n' print '6- image_rate_get \n' print '7- bit_rate_set \n' print '8- bit_rate_get \n' print '9- client_ip_set \n' print '10- client_ip_get \n' while True: x = int(raw_input("enter the number of the command: ")) if x < 1: exit() elif x == 1: print ffmpeg.start() elif x == 2: ffmpeg.stop() elif x == 3: print ffmpeg.image_size_set(640, 480) elif x == 4: print ffmpeg.image_size_get() elif x == 5: print ffmpeg.image_rate_set(15) elif x == 6: print ffmpeg.image_rate_get() elif x == 7: print ffmpeg.bit_rate_set (512) elif x == 8: print ffmpeg.bit_rate_get() elif x == 9: print ffmpeg.client_ip_set('10.156.0.12') elif x == 10: print ffmpeg.client_ip_get() else: print 'invalid'
